PIC EQU 20h

ORG 1000h
 CHAR DB "A"
      DB 13

ORG 2500h
 LETRA: INT 7
 MOV AL,20h
 OUT PIC,AL
 IRET

ORG 2000h
 MOV AX, LETRA
 MOV BX, 40
 MOV [BX], AX
 
 CLI
 MOV AL, 0FEh
 OUT PIC + 1, AL
 MOV AL, 10
 OUT PIC + 4, AL
 STI
 
 MOV BX, OFFSET CHAR
 
 RESET: MOV BYTE PTR [BX], 40h
 
 LOOP: INC CHAR
 MOV AL, 2
 MOV AH, 0
 CMP CHAR,  5Ah
 JZ RESET
 JMP LOOP
 
 INT 0
END
