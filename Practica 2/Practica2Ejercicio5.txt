ORG 1000h
 MSJ DB "INGRESE UN NUMERO: "
 MSJ2 DB "CARACTER NO VALIDO"
 FIN DB ?
 NUM DB ?

ORG 3000h
 ES_NUM: PUSH AX
 PUSH CX
 PUSH DX
 
 MOV AH,[BX]
 CMP AH,30h
 JS NOPE
 CMP AH,3Ah
 JNS NOPE
 MOV BX, 0FFh
 JMP FINR
 
NOPE: MOV BX,OFFSET MSJ2
MOV AL, OFFSET FIN - OFFSET MSJ2
 INT 7
 MOV BX, 0

FINR: POP DX
POP CX
POP AX
RET

ORG 2000h
 MOV BX,OFFSET MSJ
 MOV AL,OFFSET MSJ2 - OFFSET MSJ
 INT 7
 MOV BX,OFFSET NUM
 INT 6
 CALL ES_NUM
 INT 0
END
