ORG 1000H
CAD DB "BALLENAS DE ABUELAS"
    DB 00H

ORG 4000H
ES_VOCAL: MOV AL, 0FFH
CMP AH, 41H
JZ FIN
CMP AH, 45H
JZ FIN
CMP AH, 49H
JZ FIN
CMP AH, 4FH
JZ FIN
CMP AH, 55H
JZ FIN

CMP AH, 61H
JZ FIN
CMP AH, 66H
JZ FIN
CMP AH, 69H
JZ FIN
CMP AH, 6FH
JZ FIN
CMP AH, 75H
JZ FIN

MOV AL, 00H

FIN: RET

ORG 3000H
CONTAR_VOC: MOV CX, 0
LOOP: MOV AH, [BX]
CMP AH, 0
JZ FIN2
CALL ES_VOCAL
CMP AH, 0FFH
JNZ NO
INC CX
INC BX
NO: JMP LOOP
FIN2: RET

ORG 2000H
MOV BX, CAD
CALL CONTAR_VOC
HLT 
END
