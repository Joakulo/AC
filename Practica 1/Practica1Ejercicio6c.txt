ORG 3000H
MUL: MOV DX, 0
MOV BX, AX
MOV AX, [BX]
CMP AX, 0
JZ FIN
MOV BX, CX
MOV CX, [BX]
CMP CX, 0
JZ FIN
LOOP: ADD DX, AX
DEC CX
JNZ LOOP
FIN: RET
