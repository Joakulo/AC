ORG 1000h
BITE DB 10010100B
N DB 3h

ORG 3000h
ROTAR_IZQ: PUSH AX
PUSH BX
MOV BX,SP
ADD BX,10
MOV BX,[BX]
MOV AL,[BX]
ADD AL,AL
ADC AL,0
MOV WORD PTR [BX],AX
POP BX
POP AX
RET

ORG 3200h
ROTAR_IZQ_N: CMP CL,0
JZ FIN
DEC CL
CALL ROTAR_IZQ
JMP ROTAR_IZQ_N
FIN: RET

ORG 3400H
ROTAR_DER_N:MOV CH,8
SUB CH,CL
MOV CL,CH
CALL ROTAR_IZQ_N

ORG 2000h
MOV BX,OFFSET BITE
PUSH BX
MOV CX,0
MOV CL,N
CALL ROTAR_DER_N
HLT
END
