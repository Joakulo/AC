EOI EQU 20h
IMR EQU 21h
INT2 EQU 26h
HD EQU 40h
HE EQU 41h
 
ORG 1000H 
MSJ DB "INGENIERO"
CANT DB ?

ORG 2500H
IMPRIMIR: PUSH AX
MOV AL,[BX]
OUT HD,AL
INC BX
DEC CANT

MOV AL,20h
OUT EOI,AL
POP AX
IRET


ORG 2000H 

CLI

MOV AX,IMPRIMIR
MOV BX,40
MOV [BX],AX

MOV AL,0FBh
OUT IMR,AL
MOV AL,10
OUT INT2,AL

IN AL,HE
OR AL,80h
OUT HE,AL

MOV BX,OFFSET MSJ
MOV CANT,OFFSET CANT - OFFSET MSJ

STI

LOOP: CMP CANT,0
JNZ LOOP

INT 0 
END
